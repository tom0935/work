//EasyUI Form

//取回前執行序列
$('#formname').form('serialize');

//取回值
$('#formname').form('getValue','columnname');


$.messager.progress();	           //顯示等後進度條
$('#ff').form('submit', {
	url: '/query',
	onSubmit: function(){
		var isValid = $(this).form('validate');
		if (!isValid){
			$.messager.progress('close');	//關閉等後進度條
		}
		return isValid;	
	},
	success: function(){
		$.messager.progress('close');	  //關閉等後進度條
	}
});
